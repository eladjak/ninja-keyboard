📐 עיצוב אסימטרי — Game Feel Rules

  • Cards: border-radius שונה בכל פינה, e.g. 20px 12px 20px 12px — לא 12px אחיד
  • Score display: rotate(-1.5deg) קל + bold + glow — לא centered ב-grid רגיל

  • Character placement: לא centered — חצי מחוץ ל-frame, overlap על content

עקרון ה-Sound Design — אל תחזור על אותו צליל

הסיבה מספר 1 שמשתמשים מכבים צלילי keyboard: repetition. כל צליל צריך 3-4 variations של pitch/volume/timing. טווח קריטי ל-mobile speakers: 2kHz-5kHz (21% יותר intelligibility). Bass מתחת ל-100Hz לא נשמע דרך רמקולי נייד — הימנע. Tailored sound cues מגדילים learning speed ב-34%.

💻 Web Audio API — Sound Engine
TypeScript — src/lib/sound-engine.ts
// Ninja Keyboard Sound Engine — Web Audio API Synthesis
// No external files needed — pure synthesis

let audioCtx: AudioContext | null = null;

function getAudioCtx(): AudioContext {
  if (!audioCtx) {
    audioCtx = new (window.AudioContext ||
      (window as any).webkitAudioContext)();
  }
  return audioCtx;
}

interface ToneOptions {
  frequency: number;
  duration?: number;       // ms
  type?: OscillatorType;
  volume?: number;         // 0-1
  pitchEnd?: number;       // for pitch bend
  delay?: number;          // ms before starting
}

function playTone({
  frequency, duration = 100, type = 'sine',
  volume = 0.3, pitchEnd, delay = 0,
}: ToneOptions): void {
  const ctx = getAudioCtx();
  const osc = ctx.createOscillator();
  const gain = ctx.createGain();

  osc.connect(gain);
  gain.connect(ctx.destination);
  osc.type = type;

  const startTime = ctx.currentTime + delay / 1000;
  const endTime = startTime + duration / 1000;

  osc.frequency.setValueAtTime(frequency, startTime);
  if (pitchEnd !== undefined) {
    osc.frequency.linearRampToValueAtTime(pitchEnd, endTime);
  }

  // ADSR envelope
  gain.gain.setValueAtTime(0, startTime);
  gain.gain.linearRampToValueAtTime(volume, startTime + 0.01);
  gain.gain.exponentialRampToValueAtTime(0.001, endTime);

  osc.start(startTime);
  osc.stop(endTime + 0.05);
}

// Variation: prevents repetition fatigue
function withVariation(freq: number, range = 40): number {
  return freq + (Math.random() - 0.5) * range;
}

// ===== PUBLIC SOUND API =====

export const Sounds = {
  keystroke: (correct: boolean) => {
    if (correct) {
      playTone({ frequency: withVariation(880, 80), duration: 80, type: 'triangle', volume: 0.25 });
    } else {
      playTone({ frequency: 200, duration: 150, type: 'sawtooth', volume: 0.2, pitchEnd: 100 });
    }
  },

  wordComplete: () => {
    [523, 659, 784].forEach((freq, i) =>
      playTone({ frequency: freq, duration: 200, type: 'sine', volume: 0.25, delay: i * 70 })
    );
  },

  lessonComplete: () => {
    [523, 659, 784, 1047].forEach((freq, i) =>
      playTone({ frequency: freq, duration: 600, type: 'sine', volume: 0.2, delay: i * 100 })
    );
  },

  levelUp: () => {
    [262, 330, 392, 523, 784].forEach((freq, i) =>
      playTone({ frequency: freq, duration: 200, type: 'sine', volume: 0.4, delay: i * 120 })
    );
  },

  streakAdd: () => {
    playTone({ frequency: 880, duration: 120, type: 'triangle', volume: 0.3 });
    playTone({ frequency: 1108, duration: 200, type: 'triangle', volume: 0.3, delay: 120 });
  },

  streakBreak: () => {
    [440, 220, 110].forEach((freq, i) =>
      playTone({ frequency: freq, duration: 250, type: 'sawtooth', volume: 0.25, delay: i * 180 })
    );
  },

  button: () => playTone({ frequency: 440, duration: 60, type: 'sine', volume: 0.2 }),

  buttonAction: () => playTone({ frequency: 523, duration: 80, type: 'triangle', volume: 0.25 }),

  xpFloat: () => playTone({ frequency: 880, duration: 200, type: 'sine', volume: 0.15, pitchEnd: 1108 }),

  timerWarning: () => {
    for (let i = 0; i < 5; i++) {
      playTone({ frequency: 880, duration: 30, type: 'square', volume: 0.3, delay: i * 200 });
    }
  },

  victory: () => {
    const chord = [523, 659, 784, 1047];
    chord.forEach(f => playTone({ frequency: f, duration: 1500, type: 'sine', volume: 0.2 }));
    // Add sparkle on top
    setTimeout(() => playTone({ frequency: 2093, duration: 300, type: 'triangle', volume: 0.15 }), 500);
  },
};

'use client';
import { motion, AnimatePresence } from 'framer-motion';
import { useState, useCallback, useEffect } from 'react';

// XP floating numbers
interface FloatItem { id: number; amount: number; x: number; }

export function useXpFloat() {
  const [floats, setFloats] = useState<FloatItem[]>([]);

  const triggerXp = useCallback((amount: number) => {
    const id = Date.now() + Math.random();
    const x = 40 + Math.random() * 20; // 40-60% from left
    setFloats(prev => [...prev, { id, amount, x }]);
    setTimeout(() => setFloats(prev => prev.filter(f => f.id !== id)), 1100);
  }, []);

  return { floats, triggerXp };
}

export function XpFloatLayer({ floats }: { floats: FloatItem[] }) {
  return (
    <div className="pointer-events-none fixed inset-0 z-50" aria-hidden>
      <AnimatePresence>
        {floats.map(f => (
          <motion.div
            key={f.id}
            className="absolute font-black text-xl text-[#fdcb6e]"
            style={{ left: `${f.x}%`, bottom: '25%' }}
            initial={{ opacity: 0, y: 0, scale: 0.7 }}
            animate={{ opacity: [0, 1, 1, 0], y: -90, scale: [0.7, 1.15, 1.05, 1] }}
            exit={{}}
            transition={{ duration: 1.0, ease: 'easeOut' }}
          >
            +{f.amount} XP ⭐
          </motion.div>
        ))}
      </AnimatePresence>
    </div>
  );
}

// Celebration overlay
type CelebLevel = 'good' | 'great' | 'perfect' | 'levelup';

const CONFIGS = {
  good:    { emoji: '👍', text: 'כל הכבוד!', color: '#74b9ff' },
  great:   { emoji: '🔥', text: 'מצוין!',   color: '#00B894' },
  perfect: { emoji: '⭐', text: 'מושלם!',    color: '#6C5CE7' },
  levelup: { emoji: '🏆', text: 'עלית רמה!', color: '#fdcb6e' },
};

export function CelebrationOverlay({
  level, show, onDone,
}: { level: CelebLevel; show: boolean; onDone: () => void }) {
  const cfg = CONFIGS[level];

  useEffect(() => {
    if (show) {
      // Lazy import canvas-confetti
      import('canvas-confetti').then(({ default: confetti }) => {
        confetti({
          particleCount: level === 'levelup' ? 200 : 80,
          spread: 70,
          origin: { y: 0.55 },
        });
      });
      const t = setTimeout(onDone, 2200);
      return () => clearTimeout(t);
    }
  }, [show, level, onDone]);

  return (
    <AnimatePresence>
      {show && (
        <motion.div
          className="fixed inset-0 z-50 flex items-center justify-center pointer-events-none"
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0, transition: { delay: 1.5 } }}
        >
          <motion.div
            className="text-center"
            initial={{ scale: 0.3, y: 60 }}
            animate={{ scale: [0.3, 1.25, 1.0], y: 0 }}
            transition={{ duration: 0.5, ease: [0.34, 1.56, 0.64, 1] }}
          >
            <div className="text-8xl mb-4">{cfg.emoji}</div>
            <div
              className="text-5xl font-black"
              style={{ color: cfg.color, textShadow: `0 0 40px ${cfg.color}` }}
            >
              {cfg.text}
            </div>
          </motion.div>
        </motion.div>
      )}
    </AnimatePresence>
  );
}
