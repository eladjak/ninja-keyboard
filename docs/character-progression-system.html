// src/hooks/use-progression-check.ts
export function useProgressionCheck() {
  // Collects context from all stores:
  // xp-store: speedTestAttempts, gamesPlayed
  // practice-history: results.length, getBestWpm()
  // story-store: bossResults, unlockedCharacters

  const checkProgression = useCallback((): ProgressionEvent[] => {
    // 1. Check character unlocks → getNewlyUnlockable(ctx)
    // 2. Check badge unlocks    → getNewlyEarnedBadges(ctx)
    // 3. Check rival unlocks    → battleRecordStore.isRivalUnlocked()
    return events
  }, [/* all stores */])

  return { checkProgression }
}
