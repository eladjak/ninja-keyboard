שימוש בקוד — React + Framer Motion

דוגמאות ייבוא ושימוש ב-GameButton

TypeScript — Basic Usage
import { GameButton } from '@/components/ui/game-button'

// Primary (default) - with glow pulse
<GameButton>🥷 התחל שיעור</GameButton>

// Secondary - green
<GameButton variant="secondary">✅ המשך</GameButton>

// Danger - red
<GameButton variant="danger">❌ עצור</GameButton>

// Warning - yellow with dark text
<GameButton variant="warning">⚠️ אזהרה</GameButton>

// Ghost - transparent
<GameButton variant="ghost">👻 רפאים</GameButton>
TypeScript — Sizes
<GameButton size="sm">קטן</GameButton>
<GameButton size="md">בינוני</GameButton>  // default
<GameButton size="lg">גדול</GameButton>
<GameButton size="xl">ענק</GameButton>
TypeScript — States & Features
// Disabled
<GameButton disabled>🔒 סגור</GameButton>

// Loading
<GameButton loading>טוען...</GameButton>

// Silent (no click sound)
<GameButton silent>שקט</GameButton>

// With onClick
<GameButton
  variant="primary"
  size="lg"
  onClick={() => router.push('/lessons')}
>
  🥷 התחל ללמוד!
</GameButton>

// Custom className
<GameButton className="w-full">רוחב מלא</GameButton>
TypeScript — Component Props Interface
interface GameButtonProps {
  variant?: 'primary' | 'secondary' | 'danger' | 'warning' | 'ghost'
  size?: 'sm' | 'md' | 'lg' | 'xl'
  loading?: boolean     // Shows spinner + disables
  silent?: boolean      // Mutes click sound
  disabled?: boolean    // Grayed out, no interaction
  children: React.ReactNode
  // + all standard button props (onClick, aria-label, etc.)
}
